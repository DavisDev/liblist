688 static _list_node *listsort ( _list_node *list, _list_node ** updated_tail, comperator cmp , int flags)
689 {
690  _list_node *p, *q, *e, *tail;
691  int insize, nmerges, psize, qsize, i;
692  int direction = ( flags & SORT_LIST_ASCENDING ) ? 1 : -1;
693  insize = 1;
694 
695  while ( 1 )
696  {
697  p = list;
698  list = NULL;
699  tail = NULL;
700 
701  nmerges = 0; /* count number of merges we do in this pass */
702 
703  while ( p )
704  {
705  nmerges++; /* there exists a merge to be done */
706  /* step `insize' places along from p */
707  q = p;
708  psize = 0;
709  for ( i = 0; i < insize; i++ )
710  {
711  psize++;
712  q = q->next;
713  if ( !q )
714  {
715  break;
716  }
717  }
718 
719  /* if q hasn't fallen off end, we have two lists to merge */
720  qsize = insize;
721 
722  /* now we have two lists; merge them */
723  while ( psize > 0 || ( qsize > 0 && q ) )
724  {
725 
726  /* decide whether next element of merge comes from p or q */
727  if ( psize == 0 )
728  {
729  /* p is empty; e must come from q. */
730  e = q;
731  q = q->next;
732  qsize--;
733  }
734  else
735  if ( qsize == 0 || !q )
736  {
737  /* q is empty; e must come from p. */
738  e = p;
739  p = p->next;
740  psize--;
741  }
742  else
743  if ( ( direction * cmp ( p->node, q->node ) ) <= 0 )
744  {
745  /* First element of p is lower (or same);
746  * e must come from p. */
747  e = p;
748  p = p->next;
749  psize--;
750  }
751  else
752  {
753  /* First element of q is lower; e must come from q. */
754  e = q;
755  q = q->next;
756  qsize--;
757  }
758 
759  /* add the next element to the merged list */
760  if ( tail )
761  {
762  tail->next = e;
763  }
764  else
765  {
766  list = e;
767  }
768 
769  tail = e;
770  }
771 
772  /* now p has stepped `insize' places along, and q has too */
773  p = q;
774  }
775 
776  tail->next = NULL;
777 
778  /* If we have done only one merge, we're finished. */
779  if ( nmerges <= 1 ) /* allow for nmerges==0, the empty list case */
780  {
781  break;
782  }
783  /* Otherwise repeat, merging lists twice the size */
784  insize *= 2;
785  }
786 
787  *updated_tail = tail;
788  return list;
789 }
