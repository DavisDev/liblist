74 void llist_destroy ( llist list, bool destroy_nodes, node_func destructor )
75 {
76  _list_node * iterator;
77  _list_node * next;
78 
79  if ( list == NULL )
80  {
81  return;
82  }
83 #ifdef LLIST_OPT_SYNCHRONOUS
84  pthread_mutex_lock ( & ( ( _llist * ) list )->mutex );
85 #endif
86  // Delete the data contained in the nodes
87  iterator = ( ( _llist * ) list )->head;
88 
89  while ( iterator != NULL )
90  {
91 
92  if ( destroy_nodes )
93  {
94 
95  if ( destructor )
96  {
97  destructor ( iterator->node );
98  }
99  else
100  {
101  free ( iterator->node );
102  }
103  }
104 
105  next = iterator->next;
106 
107  free ( iterator ); // Delete's the container
108 
109  iterator = next;
110  }
111 
112  // OK, now the linked list is caput, let's delete the actual list pointer
113 #ifdef LLIST_OPT_SYNCHRONOUS
114  pthread_mutex_unlock ( & ( ( _llist * ) list )->mutex );
115  pthread_mutex_destroy ( & ( ( _llist * ) list )->mutex );
116 #endif
117  free ( list );
118 
119  return;

141 int llist_add_node ( llist list, llist_node node, int flags )
142 {
143  _list_node * node_wrapper = NULL;
144 
145  if ( list == NULL )
146  {
147  return LLIST_NULL_ARGUMENT;
148  }
149 
150  node_wrapper = malloc ( sizeof ( _list_node ) );
151 
152  if ( node_wrapper == NULL )
153  {
154  return LLIST_ERROR;
155  }
156 
157  node_wrapper->node = node;
158 #ifdef LLIST_OPT_SYNCHRONOUS
159  pthread_mutex_lock ( & ( ( _llist * ) list )->mutex );
160 #endif
161 
162  ( ( _llist * ) list )->count++;
163 
164  if (( ( _llist * ) list )->head == NULL)
165  {// Adding the first node, update head and tail to point to that node
166  node_wrapper->next = NULL;
167  ( ( _llist * ) list )->head = ( ( _llist * ) list )->tail = node_wrapper;
168  }
169  else if ( flags & ADD_NODE_FRONT )
170  {
171  node_wrapper->next = ( ( _llist * ) list )->head;
172  ( ( _llist * ) list )->head = node_wrapper;
173  }
174  else // add node in the rear
175  {
176  node_wrapper->next = NULL;
177  ( ( _llist * ) list )->tail->next = node_wrapper;
178  ( ( _llist * ) list )->tail = node_wrapper;
179  }
180 #ifdef LLIST_OPT_SYNCHRONOUS
181  pthread_mutex_unlock ( & ( ( _llist * ) list )->mutex );
182 #endif
183  return LLIST_SUCCESS;

186 int llist_delete_node ( llist list, llist_node node, equal alternative, bool destroy_node, node_func destructor )
187 {
188  _list_node * iterator;
189  _list_node * temp;
190  equal actual_equal;
191 
192  if ( ( list == NULL ) || ( node == NULL ) )
193  {
194  return LLIST_NULL_ARGUMENT;
195  }
196 
197  actual_equal = ( ( _llist * ) list )->equal_func;
198 
199  if ( alternative )
200  {
201  actual_equal = alternative;
202  }
203 
204  if ( actual_equal == NULL )
205  {
206  return LLIST_EQUAL_MISSING;
207  }
208 #ifdef LLIST_OPT_SYNCHRONOUS
209  pthread_mutex_lock ( & ( ( _llist * ) list )->mutex );
210 #endif
211  iterator = ( ( _llist * ) list )->head;
212 
213  // is it the first node ?
214  if ( actual_equal(iterator->node,node) )
215  {
216  if (destroy_node)
217  {
218  if (destructor)
219  {
220  destructor(iterator->node);
221  }
222  else
223  {
224  free(iterator->node);
225  }
226 
227  }
228 
229  ( ( _llist * ) list )->head = ( ( _llist * ) list )->head->next;
230  free ( iterator );
231  ( ( _llist * ) list )->count--;
232 
233  if (( ( _llist * ) list )->count == 0)
234  {
235  /*
236  * if we deleted the last node, we need to reset the tail also
237  * There's no need to check it somewhere else, because the last node must be the head (and tail)
238  */
239  ( ( _llist * ) list )->tail = NULL;
240  }
241  assert ( ( ( _llist * ) list )->count >= 0 );
242 #ifdef LLIST_OPT_SYNCHRONOUS
243  pthread_mutex_unlock ( & ( ( _llist * ) list )->mutex );
244 #endif
245 
246  return LLIST_SUCCESS;
247  }
248  else
249  {
250  while ( iterator->next != NULL )
251  {
252  if ( actual_equal(iterator->next->node, node) )
253  {
254  // found it
255  temp = iterator->next;
256  iterator->next = temp->next;
257  free ( temp );
258 
259  ( ( _llist * ) list )->count--;
260  assert ( ( ( _llist * ) list )->count >= 0 );
261 #ifdef LLIST_OPT_SYNCHRONOUS
262  pthread_mutex_unlock ( & ( ( _llist * ) list )->mutex );
263 #endif
264  return LLIST_SUCCESS;
265  }
266 
267  iterator = iterator->next;
268  }
269  }
270 
271  if ( iterator->next == NULL )
272  {
273 #ifdef LLIST_OPT_SYNCHRONOUS
274  pthread_mutex_unlock ( & ( ( _llist * ) list )->mutex );
275 #endif
276  return LLIST_NODE_NOT_FOUND;
277  }
278 
279  assert ( 1==2 );
280  // this assert always failed. we assume that the function never gets here...
281  return LLIST_ERROR;
282 }

321

497 int llist_concat(llist first, llist second)
498 {
499  _list_node * end_node;
500 
501  if ( (first == NULL) || (second == NULL) )
502  {
503  return LLIST_NULL_ARGUMENT;
504  }
505 
506  // Take the mutex from both the lists
507 #ifdef LLIST_OPT_SYNCHRONOUS
508  pthread_mutex_lock ( & ( ( _llist * ) first )->mutex );
509  pthread_mutex_lock ( & ( ( _llist * ) second )->mutex );
510 #endif
511 
512  end_node = ( ( _llist * ) first )->tail;
513 
514  ( ( _llist * ) first )->count += ( ( _llist * ) second )->count;
515 
516  if (end_node != NULL) // if the first list is not empty
517  {
518  end_node->next = ( ( _llist * ) second )->head;
519  }
520  else
521  { // It's empty
522  ( ( _llist * ) first )->head = ( ( _llist * ) first )->tail = ( ( _llist * ) second )->head;
523  }
524 
525  // Delete the nodes from the second list. (not really deletes them, only loses their reference.
526  (( _llist * ) second )->count = 0;
527  (( _llist * ) second )->head = (( _llist * ) second )->tail = NULL;
528 
529 #ifdef LLIST_OPT_SYNCHRONOUS
530  pthread_mutex_unlock ( & ( ( _llist * ) first )->mutex );
531  pthread_mutex_unlock ( & ( ( _llist * ) second )->mutex );
532 #endif
533 
534  return LLIST_SUCCESS;
535 }
